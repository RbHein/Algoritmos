/* 1. Escrever um algoritmo que lê 5 valores para a, um de cada vez, e conta
quantos destes valores são negativos, escrevendo esta informação. */

#include <stdio.h>
#include <stdlib.h>

int main ()
{
    int contagem = 0, num, negativos = 0;

    while (contagem <= 5)
    {
        printf("Informe um numero: ");
        scanf("%d", &num);
        contagem++;

        if (num < 0)
        {
            negativos++;
        }
    }

    printf("Quantidade de Negativos: %d", negativos);

    return 0;

}


EXERCÍCIO 2

/* 2. Escrever um algoritmo que lê um valor N inteiro e positivo e que calcula e
escreve o valor de E.
E = 1 + 1 / 1! + 1 / 2! + 1 / 3! + 1 / N!     */

#include <stdio.h>
#include <stdlib.h>

int main ()
{
    int n;
    double e = 1, fatorial = 1;

    printf("Informe o valor de N: ");
    scanf("%d", &n);

    if (n < 0)
    {
        printf("Erro. N deve ser inteiro e positivo!\n");
        return 1;
    }

    for (int i = 1; i <= n; i++)
    {
        fatorial *= i;
        e += 1 / fatorial;
    }

    printf("E: %.2f", e);
    
    return 0;

}



EXERCÍCIO 3

/* 3. A prefeitura de uma cidade fez uma pesquisa entre seus habitantes,
coletando dados sobre o salário e número de filhos. A prefeitura deseja saber:
a) média do salário da população;
b) média do número de filhos;
c) maior salário;
d) percentual de pessoas com salário até R$100,00.
O final da leitura de dados se dará com a entrada de um salário negativo. (Use
o comando ENQUANTO-FAÇA)  */

#include <stdio.h>

int main() 
{
    int nFilhos = 0, pessoas100 = 0, totalPessoas = 0;
    float salario = 0, mediaSalario = 0, mediaFilhos = 0, maiorSalario = 0;

    do 
    {
        printf("Informe o salario: ");
        scanf("%f", &salario);

        if (salario >= 0) 
        {
            printf("Informe o numero de filhos: ");
            scanf("%d", &nFilhos);

            mediaSalario = ((mediaSalario * totalPessoas) + salario) / (totalPessoas + 1);

            mediaFilhos = ((mediaFilhos * totalPessoas) + nFilhos) / (totalPessoas + 1);

            if (salario > maiorSalario) 
            {
                maiorSalario = salario;
            }

            if (salario <= 100) 
            {
                pessoas100++;
            }

            totalPessoas++;
        }

    } while (salario >= 0);

    printf("Media salarial da populacao: %.2f\n", mediaSalario);
    printf("Media do numero de filhos: %.2f\n", mediaFilhos);
    printf("Maior Salario: %.2f\n", maiorSalario);
    printf("Percentual de pessoas com salario ate R$100.00: %.2f%%\n", ((float) pessoas100 / totalPessoas) * 100);

    return 0;

}


EXERCÍCIO 4


/* 4. Chico tem 1,50 metro e cresce 2 centímetros por ano, enquanto Zé tem 1,10
metro e cresce 3 centímetros por ano. Construa um algoritmo que calcule e
imprima quantos anos serão necessários para que Zé seja maior que Chico */

#include <stdio.h>

int main ()
{
    int anos = 0;
    float chico = 1.50, ze = 1.10;

    while (chico >= ze)
    {
        chico = chico + 0.02;
        ze = ze + 0.03;
        anos++;
    }

    printf("%d anos serao necessarios para que Ze seja maior que Chico", anos);

    return 0;
}

/* Construir um algoritmo que calcule a média aritmética de vários valores
inteiros positivos, lidos externamente. O final da leitura acontecerá quando for
lido um valor negativo. */

#include <stdio.h>

int main ()
{
    int num, soma = 0, contagem = 0;
    float media;

    printf("Informe valores inteiros e positivos para calcular a Media. Numero Negativo para sair\n");

    while (1)
    {
        scanf("%d", &num);

        if (num < 0)
        {
            break;
        }

        soma += num;
        contagem++;
    }

    if (contagem > 0)
    {
        media = soma / contagem;
        printf("Media dos Valores: %.2f\n", media);
    }

    else
    {
        printf("Nao foram informados valores positivos.");
    }

    return 0;

}


EXERCÍCIO 6


/* 6. Em uma eleição presidencial existem quatro candidatos. Os votos são
informados através de códigos. Os dados utilizados para a contagem dos votos
obedecem à seguinte codificação:
- 1,2,3,4 = voto para os respectivos candidatos;
- 5 = voto nulo;
- 6 = voto em branco;
Elabore um algoritmo que leia o código do candidado em um voto. Calcule e
escreva:
- total de votos para cada candidato;
- total de votos nulos;
- total de votos em branco;
Como finalizador do conjunto de votos, tem-se o valor 0. */

#include <stdio.h>

int main ()
{
    int candidato1 = 0, candidato2 = 0, candidato3 = 0, candidato4 = 0, nulos = 0, brancos = 0, voto = -1;

    while (voto != 0)
    {
        printf("Digite o codigo do candidato, ou 5 para Voto Nulo, 6 para Voto em Branco, ou 0 para Encerrar: ");
        scanf("%d", &voto);

        switch (voto)
        {
        case 1:
            candidato1++;
            break;

        case 2:
            candidato2++;
            break;

        case 3:
            candidato3++;
            break;

        case 4:
            candidato4++;
            break;

        case 5:
            nulos++;
            break;

        case 6:
            brancos++;
            break;

        case 0:
            break;

        default:
            printf("Codigo Invalido!\n");
            break;

        }
    }

    printf("Total de Votos do Candidato 1: %d\n", candidato1);
    printf("Total de Votos do Candidato 2: %d\n", candidato2);
    printf("Total de Votos do Candidato 3: %d\n", candidato3);
    printf("Total de Votos do Candidato 4: %d\n", candidato4);
    printf("Total de Votos Nulos: %d\n", nulos);
    printf("Total de Votos em Branco: %d\n", brancos);

    return 0;
    
}

deles. Mostre o resultado. */

#include <stdio.h>

{
    int i, valores, maior, menor;

    for (i = 0; i < 50; i++)
    {
        printf("Informe um valor: ");

        {
            maior = valores;

        {

    return 0;
}

/* 10. Escreva um algoritmo que leia o código de um aluno e suas três notas.
Calcule a média ponderada do aluno, considerando que o peso para a maior
nota seja 4 e para as duas restantes, 3. Mostre o código do aluno, suas três
notas, a média calculada e uma mensagem "APROVADO" se a média for maior
ou igual a 5 e "REPROVADO" se a média for menor que 5. Repita a operação
até que o código lido seja negativo */

#include <stdio.h>
#include <stdlib.h>

int main ()
{
    int codigo;
    float n1, n2, n3, media;

    do
    {
        printf("Informe o codigo do aluno. (Valor Negativo para sair): ");

        if (codigo >= 0)
        {
            printf("Informe as tres notas do aluno, separadas por espaco: \n");
            scanf("%f %f %f", &n1, &n2, &n3);

            // Media Ponderada //

            if (n1 >= n2 && n1 >= n3)
            {
                media = (n1 * 4 + n2 * 3 + n3 * 3) / 10;
            }

            else if (n2 >= n1 && n2 >= n3)
            {
                (media = n2 * 4 + n1 * 3 + n3 * 3) / 10;
            }

            else
            {
                media = (n3 * 4 + n1 * 3 + n2 * 3) / 10;
            }

            printf("Aluno Codigo %d, notas %.1f, %.1f, %.1f, media %.2f\n", codigo, n1, n2, n3, media);

                printf("Aprovado\n");
