/* 1. Faça um programa para ler a nota da prova de 15 alunos e armazene num
vetor, calcule e imprima a média geral. */

#include <stdio.h>
#include <stdlib.h>

int main() 
{
    int i;
    float soma = 0, media, vetor[15];

    printf("Informe a nota da prova dos 15 alunos: \n");

    for (i = 0; i < 15; i++)
    {
        scanf("%f", &vetor[i]);
        soma += vetor[i];
    }
    
    media = soma / 15;

    if (media >= 0 && media <= 10)
    {
        printf("Media Geral: %.2f \n", media);
    }
    
    else
    {
        printf("Media Invalida. Tente Novamente com as notas corretas.");
    }

    return 0;
}

/* 2. Faça um programa que preencha um vetor com 10 números reais, calcule
e mostre a quantidade de números negativos e a soma dos números
positivos desse vetor. */

#include <stdio.h>
#include <stdlib.h>

int main() 
{
    int i; 
    float somaPositivos, qNegativos, vetor[10];
    
    printf("Informe 10 Numeros Reais: \n");

    for (i = 0; i < 10; i++)
    {
        scanf("%f", &vetor[i]);

        if (vetor[i] > 0)
        {
            somaPositivos += vetor[i];
        }

        else if (vetor[i] < 0)
        {
            qNegativos++;
        }
    }

    printf("Soma dos Positivos: %.2f\n", somaPositivos);
    printf("Quantidade de Negativos: %.2f\n", qNegativos);

    return 0;
}


EXERCÍCIO 3

/* Fazer um programa para ler 5 valores e, em seguida, mostrar todos os
valores lidos juntamente com o maior, o menor e a média dos valores. */

#include <stdio.h>
#include <stdlib.h>

int main() 
{
    int i, maior, menor, vetor[5];
    float soma = 0, media;

    printf("Informe 5 valores inteiros: \n");

    for (i = 0; i < 5; i++) 
    {
        scanf("%d", &vetor[i]);
        soma += vetor[i];

        if (i == 0) 
        {
            maior = vetor[i];
            menor = vetor[i];
        } 
        
        else 
        {
            if (vetor[i] > maior)
            {
                maior = vetor[i];
            }

            if (vetor[i] < menor)
            {
                menor = vetor[i];
            }
        }
    }

    media = soma / 5;

    printf("Todos os Valores Lidos: \n");

    for (i = 0; i < 5; i++) 
    {
        printf("%d ", vetor[i]);
    }

    printf("\nMaior Valor: %d \n", maior);
    printf("Menor Valor: %d \n", menor);
    printf("Media: %.2f \n", media);

    return 0;
}

/* 4. Faça um programa que leia um vetor de 5 posições para números reais e
mostre o vetor na ordem direta e depois na ordem inversa. 
*/

#include <stdio.h>
#include <stdlib.h>

int main() 
{
    int i;
    float vetor[5];

    printf("Informe 5 Valores Reais: \n");

    for (i = 0; i < 5; i++)
    {
        scanf("%f", &vetor[i]);
    }

    printf("Ordem Original (Direta) do Vetor: \n");

    for (i = 0; i < 5; i++)
    {
        printf("%.2f \n", vetor[i]);
    }

    printf("Ordem Inversa do Vetor: \n");

    for (i = 4; i >= 0; i--)
    {
        printf("%.2f \n", vetor[i]);
    }
    
    return 0;
}

/* 5. Leia um vetor de 10 posições e atribua valor 0 para todos os elementos
que possuírem valores negativos */

#include <stdio.h>
#include <stdlib.h>

int main() 
{
    int i, vetor[10];

    printf("Informe 10 Valores Inteiros: \n");

    for (i = 0; i < 10; i++)
    {
        scanf("%d", &vetor[i]);

        if (vetor[i] < 0)
        {
            vetor[i] = 0;
        }
    }

    printf("Vetor Atualizado: \n");

    for (i = 0; i < 10; i++)
    {
        printf("%d ", vetor[i]);
    }

    return 0;
}


EXERCÍCIO 6


/* 6. Faça um programa que leia um vetor de 10 números. Leia um número X.
Conte os múltiplos do número inteiro X num vetor e mostre-os na tela.  */

#include <stdio.h>
#include <stdlib.h>

int main() 
{
    int i, x, contador, vetor[10];
    
    printf("Informe 10 Valores Inteiros: \n");

    for (i = 0; i < 10; i++)
    {
        scanf("%d", &vetor[i]);
    }

    printf("Informe um valor inteiro qualquer (X): \n");
    scanf("%d", &x);

    printf("Multiplos de %d encontrados no Vetor: \n", x);

    for (i = 0; i < 10; i++)
    {
        if (vetor[i] % x == 0)
        {
            printf("%d ", vetor[i]);
            contador++;
        }
    }

    if (contador == 0)
    {
        printf("Nenhum Multiplo de %d encontrado no vetor. ", x);
    }

    return 0;
}
